CODE SEGMENT
        ASSUME CS:CODE, DS:CODE
        
        MOV AX, 32H
        MOV CX, 10H

FLIP:   SHR AX, 01H
        JC  APPEND1
        JNC APPEND0 
FLIP0:        
        LOOP FLIP 
          
        
        HLT
                    
   
APPEND0:
        SAL DX, 01H  
        JMP FLIP0
        

APPEND1:
        SAL DX, 01H
        INC DX   
        INC BX
        JMP FLIP0
        
           
    
CODE ENDS
     END
